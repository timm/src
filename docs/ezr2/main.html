def holdout(tbl):
  rows  = shuffle(tbl.rows)
  half  = len(rows)//2
  train, test = rows[:half], rows[half:]
  got   = acquire(clone(tbl, train))
  t     = tree(tbl, got)
  top   = sorted(test, key=lambda r: leaf(tbl,t,r))[:the.check]
  return min(top, key=lambda r: disty(tbl,r))
